Declaración while

...

endwhile

// Declaración Do While
do then // este ciclo while no lleva "endwhile" al final
...

Declaración for

// Puedes usar el bucle for de manera similar que en lenguajes como Java
for set i = 0 <= 10 then
...
i++
endfor

// O puedes usar una sintáxis como se usa en python, para facilitar iterar objetos o listas
for each element in list_elements then // Podrías también declarar el iterable en el mismo for. i.e: for each element in ["hola", "mundo", 1, 2, 3]
... // Este código dejará de ejecutarse cuando se iteré todo en el iterable
endfor

Declaración makeFunc

// Declarar una función es super fácil
makeFunc sum(num1, num2) then
return num1 + num2 // Las funciones pueden o no pueden retornar algo
endfunc

// Estas son funciones anonimas
(num1, num2) then // También podrías guardarla en una variable de esta forma "set var_name = () then ..."
...
endfunc(3, 7) // Cuando no guardamos una función anónima en una variable debemos llamarla después de su definición poniendo "()" después de la palabra reservada "endfunc" y si la función recibe valores pasarselos en esos paréntesis

// Así llamamos a una función después de ser definida
hola() // En el caso de que lleve argumentos se los ponemos

// Así guardamos lo que retorna una función en una variable definida
set var_name = sum(3, 7)

Clases

class Humano then
... // Puedes definir una clase y dejarla vacia con tres puntos como lo hacen otros lenguajes cuando haces un módulo y dejas una clase vacia para usarla después como clase del módulo y ponerle métodos
endclass

// Podemos heredar clases en otra clase
class Humano then
// Los métodos de las clases son como las funciones normales. Y cada clase tiene un método por defecto que se llama "init" que se ejecutará cada vez que se instancie la clase. Además, los argumentos que reciba este método, son los que se le deben pasar a la clase cuando se la instancie
makeFunc init(self, nombre, edad) then
// El argumento "self" sirve para referirse a la misma clase. O sea que si hacemos "self.nombre = nombre" estamos guardando una variable para que se pueda usar en otros métodos de la clase, por que si no tuviera el "self" solo funcionaría para el método donde está definido
self.nombre = nombre
self.edad = edad
endfunc

makeFunc caminar(self) then
return f"{self.nombre} está caminando" // Los metodos pueden o no retornar algún valor y así se formatean los strings y así se usan las variables de las clases en otros métodos que no es donde se definió
endfunc
endclass

class Trabajador inherit Humano then // Y con la palabra reservada "inherit" heredas todo lo que está en la clase que hereda, tanto métodos como variables declaradas con "self"
...
endclass

// Intanciamos una clase de la siguiente manera
set enrique = Humano("Enrique", 31)
// Y podemos usar sus métodos así:
enrique.caminar() // El argumento "self" no se pasa cuando llamas a un método de clase, porque lo toma automaticamente de la clase instanciada

Interfaces

// Las interfaces son básicamente una lista de acciones que la clase, o objeto que lo herede debe cumplir
interface IHumano then // Los nombres de las intefaces suelen tener una "I" al principio del nombre, esto es solo por convención
// En las interfaces no haremos más que solo definir lo que van a necesitar las clases u objetos que la vayan a heredar, veamos un ejemplo
nombre: String // Esto básicamente crea un objeto que su identificador va a ser "Humano" y lo que va a contener es { nombre: String } y el string se lo pasaremos cuando la heredemos en alguna clase
edad: {
type: Integer,
required: false
} // Podemos definir objetos en una propiedad de una interfáz para decir si es obligatorio ponerla y el tipo que requiere
endinterface

class Empleado inherit IHumano then
makeFunc init(self, nombre) then
self.nombre = nombre // Aquí no estamos creando una variable, sino que le estamos dando un valor al indentificador de la interfáz que si o si tiene que ser un string y que es obligatorio definirlo
endfunc
endclass

// Entonces si queremos conseguir ese atributo que definimos dentro de la clase de la interfaz hacemos esto
set pepe = Empleado("Pepe")
print(pepe.nombre) // print es una función built-in que sirve para inprimir en consola lo que se le pase por argumento. en este caso imprime "Pepe" que es el valor que tiene ese atributo
// Podemos definir algún atributo no requerido de la interfáz desde fuera de la clase
pepe.edad = 42 // Y ahora el valor de edad de la interfáz que antes era "null" ahora es "42"
