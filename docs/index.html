Operaciones con strings

"Hola " + "mundo" // devuelve Hola mundo

Operaciones con números enteros

1 + 1 // devuelve 2
1 - 1 // devuelve 0
2 * 2 // devuelve 4
4 / 2 // devuelve 2
!-5 // devuelve 5
!5 // devuelve -5

Operaciones con números flotantes

1.5 + 1.5 // devuelve 3
1.5 - 0.5 // devuelve 1.0
.3 * 2.5 // devuelve 0.75
4.1 / 2.33 // devuelve 1.759656652
!-5.5 // devuelve 5.5
!5.5 // devuelve -5.5

Operaciones con booleanos

true == true // devuelve true
false != true // devuelve true
!true // devuelve false
!!false // devuelve true

Operaciones condicionales

5 == 5 // devuelve true
4 < 19 // devuelve true
50 > 0.4 // devuelve true
2 != 2 // devuelve false

Definir variables

set num = 34
set num // Puedes definir una variable sin valor
num = 34 // Y luego puedes definirle el valor de esta forma
set num: Integer = 34 // También puedes definir la variable con un tipo ya definido

Declaración if

// Declaración if
if 5 == 5 then
...
endif

// Declaración if sin condicional
if true then
...
endif

// Negando el booleano
if !false then
...
endif

// Declaración if elseif y else
if false then
...
elseif false then
...
else
...
endif

Declaración while

// Declaración while
while true == true do
...
break // Usamos "break" para terminar ciclos while a no ser que quieras que la condicional en algún momento sea false
endwhile

// Declaración Do While
do // este ciclo while no lleva "endwhile" al final
...
num++ // podemos sumar un número a una variable así
while set num = 0 < 10 // Podemos declarar variables solo para el bucle

Declaración for

// Puedes usar el bucle for de manera similar que en lenguajes como Java
for set i = 0 <= 10
...
i++
endfor

// O puedes usar una sintáxis como se usa en python, para facilitar iterar objetos o listas
for each element in list_elements // Podrías también declarar el iterable en el mismo for. i.e: for each element in ["hola", "mundo", 1, 2, 3]
... // Este código dejará de ejecutarse cuando se iteré todo en el iterable
endfor

Declaración makeFunc

// Declarar una función es super fácil
makeFunc sum(num1, num2) then
return num1 + num2 // Las funciones pueden o no pueden retornar algo
endfunc

// Estas son funciones anonimas
(num1, num2) then // También podrías guardarla en una variable de esta forma set var_name = () then ...
...
endfunc(3, 7) // Cuando no guardamos una función anónima en una variable debemos llamarla después de su definición poniendo "()" después de la palabra reservada "endfunc" y si la función recibe valores pasarselos en esos paréntesis

// Así llamamos a una función después de ser definida
hola() // En el caso de que lleve argumentos se los ponemos

// Así guardamos lo que retorna una función en una variable definida
set var_name = sum(3, 7)
